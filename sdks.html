Quick Start Examples

// TypeScript / JavaScript import { NexusClient } from '@nexusapi/sdk'; const nexus = new NexusClient({ apiKey: process.env.NEXUS_API_KEY }); // Get current user const user = await nexus.users.me(); // Create a project const project = await nexus.projects.create({ name: 'My New Project', description: 'A sample project' });
# Python from nexusapi import NexusClient import os client = NexusClient(api_key=os.environ.get('NEXUS_API_KEY')) # Get current user user = client.users.me() # Create a project project = client.projects.create( name='My New Project', description='A sample project' )
// Go package main import ( "fmt" "os" nexusapi "github.com/nexusapi/nexus-go" ) func main() { client := nexusapi.NewClient(os.Getenv("NEXUS_API_KEY")) user, _ := client.Users.Me() fmt.Println(user.Name) project, _ := client.Projects.Create(&nexusapi.ProjectInput{ Name: "My New Project", }) fmt.Println(project.ID) }
